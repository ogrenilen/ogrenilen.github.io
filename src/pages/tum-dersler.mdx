import sidebars from "../../sidebars";
import Link from "@docusaurus/Link";

export function ListLessons() {
    return (
        <table>
            <thead>
            <tr>
                <th>Dersin Adı</th>
            </tr>
            </thead>
            <tbody>
            {
                Object.values(sidebars).map((value, index) => (
                    <tr key={index}>
                        <td>
                            <Link to={`/docs/${value[0].dirName}/giris`}>
                                {value[0].dirName.replaceAll('-', ' ').toUpperCase()}
                            </Link>
                        </td>
                    </tr>
                ))
            }
            </tbody>
        </table>
    );
}
